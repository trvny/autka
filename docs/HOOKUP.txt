============================

1) ListingsScreen.kt — imports
   Add:
       import com.autka.feature.external.MarketplaceLinksRow

2) ListingsScreen.kt — show the row in BOTH the empty state and below results.
   Replace the existing `when { ... }` block with:

            when {
                uiState.isRefreshing && uiState.offers.isEmpty() -> LoadingIndicator()
                uiState.offers.isEmpty() -> Column(
                    verticalArrangement = Arrangement.spacedBy(16.dp),
                ) {
                    EmptyState(
                        if (uiState.activeFilterCount > 0) stringResource(R.string.empty_no_match)
                        else stringResource(R.string.empty_no_offers),
                    )
                    // No results in our cache — let the user jump straight to the
                    // marketplaces' own search, pre-filled from their filter.
                    MarketplaceLinksRow(filter = uiState.filter)
                }
                else -> LazyColumn(
                    contentPadding = PaddingValues(16.dp),
                    verticalArrangement = Arrangement.spacedBy(12.dp),
                ) {
                    items(uiState.offers, key = { it.id }) { offer ->
                        OfferCard(
                            offer = offer,
                            displayCurrency = uiState.displayCurrency,
                            rates = uiState.exchangeRates,
                            onClick = { onOfferClick(offer.id) },
                        )
                    }
                    // Footer: always offer "continue on Otomoto/OLX" under the list.
                    item {
                        MarketplaceLinksRow(
                            filter = uiState.filter,
                            modifier = Modifier.padding(top = 4.dp),
                        )
                    }
                }
            }

   (uiState.filter already exists on ListingsUiState; no ViewModel change needed.)
